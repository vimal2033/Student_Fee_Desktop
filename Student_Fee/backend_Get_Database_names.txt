// This coded grabs names and url of all the GOOGLE_SHEETS present in the same folder and send as api

// Replace this with your actual API key
const API_KEY = "api-key"; //enter api key used in front end
function doGet(e) {
    if (!e || e.parameter.apiKey !== API_KEY ) {
    return ContentService.createTextOutput(
      JSON.stringify({ status: 401, error: "Unauthorized - Invalid API Key" })
    ).setMimeType(ContentService.MimeType.JSON);
  }

  const result = getSheetsInScriptFolder();
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function getSheetsInScriptFolder() {
  const scriptFile = DriveApp.getFileById(ScriptApp.getScriptId());
  const folders = scriptFile.getParents();

  if (!folders.hasNext()) {
    return { error: "Script file is not in a folder." };
  }

  const folder = folders.next();
  const files = folder.getFilesByType(MimeType.GOOGLE_SHEETS);

  const sheets = [];

  while (files.hasNext()) {
    const file = files.next();
    sheets.push({
      name: file.getName(),
      url: file.getUrl()
    });
  }

  return { sheets };
}
